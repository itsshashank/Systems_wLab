`DECLARE
  FIRST NUMBER;
   SECOND NUMBER;
   BIG NUMBER;
   SMALL NUMBER;
   I NUMBER;
BEGIN
   FIRST:=&NUMBER;
   SECOND:=&NUMBER;
    IF(FIRST>SECOND) THEN
	   BIG:=FIRST-1;
	   SMALL:=SECOND+1;
	ELSE
	    BIG:=SECOND-1;
		SMALL:=FIRST+1;
	END IF; 
	FOR I IN SMALL..BIG LOOP
      DBMS_OUTPUT.PUT_LINE(I);
    END LOOP;
END;	
/
