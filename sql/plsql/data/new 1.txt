DECLARE
	A NUMBER;
	S NUMBER;
	BIG NUMBER;
	SMALL NUMBER;
BEGIN
     DBMS_OUTPUT.PUT_LINE('ENTER A NUMBER');
	 A:=&NUMBER;
	 DBMS_OUTPUT.PUT_LINE('ENTER A NUMBER');
	 S:=&NUMBER;
	 IF (A>S) THEN
	    BIG := A ;
		SMALL := S ;
	 ELSIF(A<S) THEN	
         BIG := S;
		 SMALL :=A;
	 ELSE
	     DBMS_OUTPUT.PUT_LINE('TWO NUMBERS ARE EQUAL');
     END IF ;		
	 
	 DBMS_OUTPUT.PUT_LINE('BIG IS ' || BIG);
	 DBMS_OUTPUT.PUT_LINE('SMALL IS '|| SMALL);
END;
